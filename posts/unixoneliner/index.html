UNIX/Linux One Liner for Bioinformatics

August 15, 2016 7-minute read
Programming • Bash

Some of the handy UNIX/Linux one liner for bioinformatics.

# Find all the file name with specific pattern and copy to a folder
find data/ -type f -name "S[0-9]*.tsv" | xargs -i cp {} sparccOutput/

# Find any files with particular pattern inside all directories
grep --recursive --ignore-case --files-with-matches "CANCELLED" Root_RST-04-106/

# Count the number of files within a folder.
ls -F |grep -v / | wc -l

#  List just the names of matching files
grep -l 'main' *.c

# Search directories recursively. Search for all the files under the directory  '/home/Rootstocks'
grep -r 'Maxifort' /home/Rootstocks

# Output context around the matching lines. Example- display header and first line of sequence for each header. 
grep -C 2 '^>' *
# AND grep
grep 'apple' /home/test | grep 'ball'

# Find the read length in fastq.gz file

zcat file.fastq.gz | awk '{if(NR%4==2) print length($1)} ' | head -1


# Find the number of reads in fastq.gz file
zgrep "^@" file.fastq.gz | wc -l

# Find the number of reads in all the fastq.gz files within a folder
for i in `ls *.fastq.gz`; do echo $(zcat ${i} | wc -l)/4|bc; done

## count the number of read in bam file
samtools view -c R8S8.bt2.bam

samtools view -c -f 4 R8S8.bt2.bam

## calculate the size of a folder
du -a Reisch-Strat-Pool-1_S1/ | cut -f1 | awk '{s+=$1} END {print s}'
du -a Reisch-Strat-Pool-1_S1/| cut -f1 | awk '{s+=$1} END {print s}'

# As you can see in the above output, the size of the directories is displayed in bits. 
# We can also display the size in human readable format.
du -ah Reisch-Strat-Pool-1_S1

# We can also display the disk usage size only in KB, or MB, or GB.
# To do so, use -k for kilobytes, -m for megabytes

du -k Reisch-Strat-Pool-1_S1/
du -m Reisch-Strat-Pool-1_S1/

# To display the disk usage of all items including files and directories, use -a flag.
du -ah Reisch-Strat-Pool-1_S1/

# We can also display the size of multiple directories at once as shown below.

du -h Reisch-Strat-Pool-1_S1 Reisch-Strat-Pool-1_S2 Reisch-Strat-Pool-1_S3

# If you want to check the total disk space used by a particular directory, use the -s flag.
du -sh Reisch-Strat-Pool-1_S1

# To display the grand total of directories, add -c flag with du -sh command.
du -ch Downloads/ | grep total

# Take in the file name and create a same diretory and the move the file into the created folder.
#!/bin/bash
for file in *; do 
	mkdir -p -- ${file%.*} 
	mv -- $file ${file%.*} 
done

awk - allows to filter, sort, and subset in linux.

# select the entry such that the value of in the third column is greater than 10
awk '$3 > 10' test.txt

## AND awk

awk '$3 > 10 && $4 > 100' test.txt

## apply mathmatical function

awk '$3 > 10 && $1 > $2*100' test.txt

## sum the length of a column

awk '{countliner += $1} END {print countliner}' test.txt

Note: Here countliner is a just a variable to store the value.

# # if you want to write to a new file
# note file name must be change to different extension -otherfile it will create a non terminating condition.

#!/bin/bash
for file in $( ls *.fna);do
     echo file: $file
     fname=$(basename $file .fna)
     echo basename: $fname
     awk '/>/{sub(">","&"FILENAME"_");sub(/\.fasta/,x)}1' $file > $fname.fasta
done
